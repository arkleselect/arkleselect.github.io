展开代码 @RestController @RequestMapping("/customer/user") @Api(tags = "小程序端用户相关接口") public class CustomerUserTypeController { @Autowired private CustomerUserService customerUserService; @PostMapping("/login") public ResponseResult login(@RequestBody UserLoginRequestDto userLoginRequestDto) { LoginVo loginVo = customerUserService.login(userLoginRequestDto); return ResponseResult.success(loginVo); } }

业务层

CustomerUserService写了一个login方法，返回Vo对象

展开代码 public interface CustomerUserService { LoginVo login(UserLoginRequestDto userLoginRequestDto); }

他的实现类CustomerUserServiceImpl

这里数组是用来登陆后选择默认的名称+手机号后四位，登录的方法一共分为了六步：

1.通过用户的code 请求微信 获取openId
2.根据openId 查询客户信息
3.判断客户信息是否为空，如果为空 重新构建（赋值openId）
4.通过phoneCode 请求微信 获取真正的手机号码
5.新增或者修改用户
6.生成token（包含用户的ID 和昵称）

（与上步骤对不起来就看着图）

用login获取用户的code，通过request发送code到开发者服务器，开发者服务器拿到code后，拿着code加上appid和密钥请求微信服务器，微信服务器返回key和openId，这里openId就是用户的唯一凭证 拿到openId后根据这个id查询用户信息 如果没有的话就注册一个，并把openId存储进去(这里引出了Builder注解)，通过phoneCode 请求微信 获取真正的手机号码，然后就是携带登录状态新增或者修改用户，并生成token（包含用户的ID 和昵称)存入storage，phondeCode可以理解为开发者服务器内部的额外操作

展开代码 @Service public class CustomerUserServiceImpl implements CustomerUserService { @Autowired private WeChatService weChatService; @Autowired private MemberMapper memberMapper; @Autowired private JwtTokenManagerProperties jwtTokenManagerProperties; static ArrayList DEFAULT_NICKNAME_PREFIX = Lists.newArrayList( "生活更美好", "大桔大利", "日富一日", "好柿开花", "柿柿如意", "一椰暴富", "大柚所为", "杨梅吐气", "天生荔枝" ); @Override public LoginVo login(UserLoginRequestDto userLoginRequestDto) { // 1. 通过用户的code 请求微信 获取openId String openId = weChatService.getOpenId(userLoginRequestDto.getCode()); // 2. 根据openId 查询客户信息 Member member = memberMapper.findByOpenId(openId); // 3. 判断客户信息是否为空，如果为空 重新构建（赋值openId） if (ObjectUtil.isEmpty(member)) { /* Member member1 = new Member(); member1.setOpenId(openId); */ // 第二种方法 需实体类加@Builder member = Member.builder().openId(openId).build(); } // 4. 通过phoneCode 请求微信 获取真正的手机号码 String phone = weChatService.getPhone(userLoginRequestDto.getPhoneCode()); // 5. 新增或者修改用户 saveOrUpdate(member, phone); // 6. 生成token（包含用户的ID 和昵称 HashMap claims = new HashMap<>(); claims.put(Constants.JWT_USERID, member.getId()); claims.put(Constants.JWT_USERNAME, member.getName()); String token = JwtUtil.createJWT( jwtTokenManagerProperties.getBase64EncodedSecretKey(), jwtTokenManagerProperties.getTtl(), claims); LoginVo loginVo = new LoginVo(); loginVo.setToken(token); loginVo.setNickName(member.getName()); return loginVo; } public void saveOrUpdate(Member member, String phone) { // 判断手机号是否需要更换 if (!phone.equals(member.getPhone())) { member.setPhone(phone); } // 判断是否需要更新 if (ObjectUtil.isNotEmpty(member.getId())) { memberMapper.update(member); } // 昵称 随机成语 + 手机号后四位 Random random = new Random(); int index = random.nextInt(DEFAULT_NICKNAME_PREFIX.size()); String nickName = DEFAULT_NICKNAME_PREFIX.get(index) + phone.substring(phone.length() - 4); member.setName(nickName); // 新增 memberMapper.insert(member); } }

这里实现类中进行的操作，使用到的方法，是单独写了一个WeChatServiceImpl实现类来进行统一操作

展开代码 @Service public class WeChatServiceImpl implements WeChatService { // 登录 private static final String REQUEST_URL = "https://api.weixin.qq.com/sns/jscode2session?grant_type=authorization_code"; // 获取token private static final String TOKEN_URL = "https://api.weixin.qq.com/cgi-bin/token?grant_type=client_credential"; // 获取手机号 private static final String PHONE_REQUEST_URL = "https://api.weixin.qq.com/wxa/business/getuserphonenumber?access_token="; @Override public String getOpenId(String code) { Map map = getAppConfig(); map.put("js_code", code); map.put("grant_type", "authorization_code"); String result = HttpUtil.get(REQUEST_URL, map); JSONObject jsonObject = JSONUtil.parseObj(result); if (ObjectUtil.isNotEmpty(jsonObject.getInt("errcode"))) { throw new RuntimeException(jsonObject.getStr("errmsg")); } String openid = jsonObject.getStr("openid"); return openid; } @Value("${zzyl.wechat.appId}") private String appId; @Value("${zzyl.wechat.appSecret}") private String appSecret; public Map getAppConfig() { Map map = new HashMap<>(); map.put("appid", appId); map.put("secret", appSecret); return map; } @Override public String getPhone(String phone) { // 1. 获取token String token = getToken(); String url = PHONE_REQUEST_URL + token; // 2. 存放参数 HashMap map = new HashMap<>(); map.put("code", phone); // 3. 发送请求 String result = HttpUtil.post(url, JSONUtil.toJsonStr(map)); JSONObject jsObject = JSONUtil.parseObj(result); if (jsObject.getInt("errcode") != 0) { throw new RuntimeException(jsObject.getStr("errmsg")); } return jsObject.getJSONObject("phone_info").getStr("phoneNumber"); } private String getToken() { Map map = getAppConfig(); String result = HttpUtil.get(TOKEN_URL, map); JSONObject jsonObject = JSONUtil.parseObj(result); if(ObjectUtil.isNotEmpty(jsonObject.getInt("errcode"))){ throw new RuntimeException(jsonObject.getStr("errmsg")); } return jsonObject.getStr("access_token"); } }

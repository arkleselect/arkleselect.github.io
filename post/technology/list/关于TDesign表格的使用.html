插槽

可以通过插槽拿到一个字段的一行数据，然后单独处理这一行的数据，以下处理三个字段的显示问题

处理图片显示

    <!-- 处理图片 -->
    <template #image="{row}">
      <t-image src="row.image" :style="{ width: '120px', height: '120px' }" />
    </template>

处理序号递增


    <!-- 处理序号 -->
    <template #rowIndex="{rowIndex}">
      {{rowIndex + 1}}
    </template>

处理性别显示

      <!-- 处理性别显示 -->
      <template #gender="{row}">
        {{ row.gender == 1 ? '男' : '女' }}
      </template>
